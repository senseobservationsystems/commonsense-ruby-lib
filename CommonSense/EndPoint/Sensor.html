def to_parameters
  param = self.to_h(false)
  if param[:data_type] == "json"
    if param[:data_structure] && !param[:data_structure].kind_of?(String)
      param[:data_structure] = param[:data_structure].to_json
    end
  end

  {sensor: param}
end
